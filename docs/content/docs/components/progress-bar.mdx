## Usage

```tsx title="app/page.tsx"
"use client";

import dynamic from "next/dynamic";

const Terminal = dynamic(() => import("./terminal"), {
  ssr: false,
});

export default function Home() {
  return <Terminal />;
}
```

```tsx title="app/terminal.tsx"
"use client";

import { InkXterm, Box, Text } from "ink-web/bundled";
import { ProgressBar } from '@/components/ui/progress-bar'
import "ink-web/bundled/css";
import "xterm/css/xterm.css";

export default function Terminal() {
  return (
    <InkXterm focus>
      <Box flexDirection="column">
        <ProgressBar percent={0.7} color="green" />
      </Box>
    </InkXterm>
  )
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `percent` | `number` | `0` | The percentage of progress (between 0 and 1) |
| `left` | `number` | `0` | Number of characters to subtract from the left side |
| `right` | `number` | `0` | Number of characters to subtract from the right side |
| `columns` | `number` | `process.stdout.columns \|\| 80` | Override the terminal width in columns |
| `character` | `string` | `'█'` | The character to use for each step of the progress bar |
| `rightPad` | `boolean` | `false` | Whether to pad the right side with spaces to fill available width |
| `color` | `string` | - | Color of the progress bar (from Ink's color palette) |

## Examples

### Basic Progress Bar

```tsx
<ProgressBar percent={0.5} />
```

### Colored Progress

Add color to indicate status:

```tsx
<Box flexDirection="column" gap={1}>
  <ProgressBar percent={0.3} color="red" />
  <ProgressBar percent={0.6} color="yellow" />
  <ProgressBar percent={1.0} color="green" />
</Box>
```

### With Percentage Label

Display progress with a percentage label:

```tsx
<Box>
  <Text>Progress: </Text>
  <ProgressBar percent={progress} left={11} right={6} color="cyan" />
  <Text> {Math.round(progress * 100)}%</Text>
</Box>
```

### Custom Character

Use different characters for varied appearances:

```tsx
<Box flexDirection="column" gap={1}>
  <ProgressBar percent={0.7} character="=" />
  <ProgressBar percent={0.7} character="▓" />
  <ProgressBar percent={0.7} character="●" />
  <ProgressBar percent={0.7} character="■" />
</Box>
```

### With Right Padding

Use `rightPad` to make the bar always fill the available width:

```tsx
<Box borderStyle="round" padding={1}>
  <ProgressBar 
    percent={0.4} 
    character="█" 
    color="blue"
    rightPad 
  />
</Box>
```

### Loading Animation

Animate progress over time:

```tsx
import { useState, useEffect } from 'react'

function LoadingBar() {
  const [progress, setProgress] = useState(0)

  useEffect(() => {
    const timer = setInterval(() => {
      setProgress(prev => {
        if (prev >= 1) return 0
        return prev + 0.01
      })
    }, 50)

    return () => clearInterval(timer)
  }, [])

  return (
    <Box flexDirection="column">
      <Text>Loading...</Text>
      <ProgressBar percent={progress} color="green" />
    </Box>
  )
}
```

### Multiple Progress Bars

Display multiple progress indicators:

```tsx
<Box flexDirection="column" gap={1}>
  <Box>
    <Text>CPU:  </Text>
    <ProgressBar percent={0.45} left={6} color="cyan" />
  </Box>
  <Box>
    <Text>RAM:  </Text>
    <ProgressBar percent={0.78} left={6} color="yellow" />
  </Box>
  <Box>
    <Text>Disk: </Text>
    <ProgressBar percent={0.23} left={6} color="green" />
  </Box>

```

### Download Progress

Show file download progress:

```tsx
function DownloadProgress({ filename, percent, size }) {
  return (
    <Box flexDirection="column">
      <Text>Downloading: {filename}</Text>
      <Box>
        <ProgressBar 
          percent={percent} 
          character="▓"
          color={percent < 1 ? "cyan" : "green"}
          rightPad
        />
      </Box>
      <Text dimColor>
        {Math.round(percent * size)}MB / {size}MB ({Math.round(percent * 100)}%)
      </Text>
    </Box>
  )
}
```

### Custom Width

Override the terminal width for specific layouts:

```tsx
<Box flexDirection="column">
  <ProgressBar percent={0.6} columns={40} />
  <ProgressBar percent={0.6} columns={20} />
</Box>
```

### With Border

Combine with Box borders for a contained look:

```tsx
<Box borderStyle="round" borderColor="cyan" padding={1}>
  <Box flexDirection="column" gap={1}>
    <Text bold color="cyan">Installation Progress</Text>
    <ProgressBar percent={0.65} character="▓" color="cyan" />
    <Text dimColor>Installing dependencies... 65%</Text>
  </Box>
</Box>
```

## Notes

- The progress bar automatically adapts to terminal width using `process.stdout.columns`
- The `percent` prop is clamped between 0 and 1 to prevent overflow
- Use `left` and `right` props to reserve space for text on the same line
- The `rightPad` prop is useful when you want a consistent width regardless of progress
- All standard Ink `Text` props (like `color`, `bold`, `dimColor`) can be used
- The component works seamlessly with Ink's layout system (`Box`, `flexDirection`, etc.)
